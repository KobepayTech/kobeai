🔥 STEP 2 — CUSTOM INSTALLER (KOBE ENGINE)

This is your installer script:

install.py

import os

print("🚀 Kobe System Installer Starting...")

# 1. Update system
os.system("apt update -y")

# 2. Install dependencies
os.system("apt install -y python3 python3-pip nginx sqlite3")

# 3. Install Python backend libs
os.system("pip3 install fastapi uvicorn requests")

# 4. Copy system files
os.system("mkdir -p /opt/kobe")
os.system("cp -r /kobe/system/* /opt/kobe/")

print("📦 System files deployed")

# 5. Setup database
os.system("sqlite3 /opt/kobe/kobe.db < /kobe/config/schema.sql")

# 6. Install services
os.system("cp /kobe/scripts/*.service /etc/systemd/system/")
os.system("systemctl daemon-reload")

# 7. Enable services
os.system("systemctl enable kobe-backend")
os.system("systemctl enable kobe-print")

# 8. Start services
os.system("systemctl start kobe-backend")
os.system("systemctl start kobe-print")

print("✅ Kobe System Installed Successfully")
