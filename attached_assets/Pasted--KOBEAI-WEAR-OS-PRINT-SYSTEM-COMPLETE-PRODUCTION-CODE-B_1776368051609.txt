```kotlin
package com.kobeai.print

import android.app.PendingIntent
import android.content.Intent
import android.nfc.NfcAdapter
import android.nfc.Tag
import android.nfc.tech.Ndef
import android.os.Bundle
import android.widget.Toast
import androidx.activity.ComponentActivity
import androidx.activity.compose.setContent
import androidx.compose.runtime.*
import androidx.lifecycle.lifecycleScope
import androidx.navigation.compose.NavHost
import androidx.navigation.compose.composable
import androidx.navigation.compose.rememberNavController
import com.kobeai.print.nfc.NfcReader
import com.kobeai.print.ui.screens.*
import com.kobeai.print.ui.theme.KobePrintTheme
import com.kobeai.print.data.PreferencesManager
import com.kobeai.print.network.NetworkModule
import dagger.hilt.android.AndroidEntryPoint
import kotlinx.coroutines.launch
import javax.inject.Inject

@AndroidEntryPoint
class MainActivity : ComponentActivity() {
    
    @Inject lateinit var prefsManager: PreferencesManager
    @Inject lateinit var networkModule: NetworkModule
    @Inject lateinit var nfcReader: NfcReader
    
    private var nfcAdapter: NfcAdapter? = null
    private var pendingIntent: PendingIntent? = null
    
    private var studentId by mutableStateOf<String?>(null)
    private var studentName by mutableStateOf<String?>(null)
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // Initialize NFC
        nfcAdapter = NfcAdapter.getDefaultAdapter(this)
        pendingIntent = PendingIntent.getActivity(
            this, 0,
            Intent(this, javaClass).addFlags(Intent.FLAG_ACTIVITY_SINGLE_TOP),
            PendingIntent.FLAG_MUTABLE
        )
        
        // Check if already logged in
        studentId = prefsManager.getStudentId()
        studentName = prefsManager.getStudentName()
        
        setContent {
            KobePrintTheme {
                val navController = rememberNavController()
                val isLoggedIn = studentId != null
                
                NavHost(
                    navController = navController,
                    startDestination = if (isLoggedIn) "home" else "login"
                ) {
                    composable("login") {
                        LoginScreen(
                            onNfcDetected = { tag ->
                                handleNfcLogin(tag)
                            },
                            studentId = studentId
                        )
                    }
                    
                    composable("home") {
                        HomeScreen(
                            studentName = studentName ?: "Student",
                            onNavigateToFiles = { category ->
                                navController.navigate("files/$category")
                            },
                            onNavigateToRecent = { navController.navigate("recent") },
                            onNavigateToFavorites = { navController.navigate("favorites") },
                            onNavigateToSettings = { navController.navigate("settings") }
                        )
                    }
                    
                    composable("files/{category}") { backStackEntry ->
                        val category = backStackEntry.arguments?.getString("category") ?: "all"
                        FileListScreen(
                            category = category,
                            studentId = studentId!!,
                            onFileSelected = { document ->
                                navController.navigate("confirm/${document.id}")
                            },
                            onBack = { navController.popBackStack() }
                        )
                    }
                    
                    composable("recent") {
                        RecentScreen(
                            studentId = studentId!!,
                            onFileSelected = { document ->
                                navController.navigate("confirm/${document.id}")
                            },
                            onBack = { navController.popBackStack() }
                        )
                    }
                    
                    composable("favorites") {
                        FavoritesScreen(
                            studentId = studentId!!,
                            onFileSelected = { document ->
                                navController.navigate("confirm/${document.id}")
                            },
                            onBack = { navController.popBackStack() }
                        )
                    }
                    
                    composable("confirm/{documentId}") { backStackEntry ->
                        val documentId = backStackEntry.arguments?.getString("documentId") ?: return@composable
                        ConfirmScreen(
                            documentId = documentId,
                            studentId = studentId!!,
                            onPrintSuccess = {
                                navController.navigate("home") {
                                    popUpTo("home") { inclusive = true }
                                }
                            },
                            onCancel = { navController.popBackStack() }
                        )
                    }
                    
                    composable("settings") {
                        SettingsScreen(
                            onBack = { navController.popBackStack() },
                            onLogout = {
                                prefsManager.clear()
                                studentId = null
                                studentName = null
                                navController.navigate("login") {
                                    popUpTo("home") { inclusive = true }
                                }
                            }
                        )
                    }
                }
            }
        }
    }
    
    override fun onResume() {
        super.onResume()
        // Enable foreground dispatch for NFC
        nfcAdapter?.enableForegroundDispatch(this, pendingIntent, null, null)
    }
    
    override fun onPause() {
        super.onPause()
        nfcAdapter?.disableForegroundDispatch(this)
    }
    
    override fun onNewIntent(intent: Intent) {
        super.onNewIntent(intent)
        
        if (NfcAdapter.ACTION_NDEF_DISCOVERED == intent.action ||
            NfcAdapter.ACTION_TAG_DISCOVERED == intent.action ||
            NfcAdapter.ACTION_TECH_DISCOVERED == intent.action) {
            
            val tag = intent.getParcelableExtra<Tag>(NfcAdapter.EXTRA_TAG)
            tag?.let { handleNfcLogin(it) }
        }
    }
    
    private fun handleNfcLogin(tag: Tag) {
        lifecycleScope.launch {
            val nfcData = nfcReader.readNfcTag(tag)
            
            if (nfcData != null && nfcData.containsKey("student_id")) {
                val id = nfcData["student_id"]!!
                
                // Authenticate with backend
                try {
                    val response = networkModule.apiService.authenticateNfc(id)
                    if (response.success) {
                        studentId = id
                        studentName = response.studentName
                        
                        prefsManager.setStudentId(id)
                        prefsManager.setStudentName(response.studentName)
                        prefsManager.setAuthToken(response.token)
                        
                        // Vibrate to confirm
                        vibrate()
                        
                        Toast.makeText(
                            this@MainActivity,
                            "Welcome, ${response.studentName}!",
                            Toast.LENGTH_SHORT
                        ).show()
                    }
                } catch (e: Exception) {
                    Toast.makeText(
                        this@MainActivity,
                        "Authentication failed",
                        Toast.LENGTH_SHORT
                    ).show()
                }
            }
        }
    }
    
    private fun vibrate() {
        val vibrator = getSystemService(VIBRATOR_SERVICE) as android.os.Vibrator
        if (vibrator.hasVibrator()) {
            vibrator.vibrate(android.os.VibrationEffect.createOneShot(100, 50))
        }
    }
}
```

📱 PART 3: NFC READER

nfc/NfcReader.kt

```kotlin
package com.kobeai.print.nfc

import android.nfc.Tag
import android.nfc.tech.Ndef
import android.util.Log
import javax.inject.Inject
import javax.inject.Singleton

@Singleton
class NfcReader @Inject constructor() {
    
    companion object {
        private const val TAG = "NfcReader"
    }
    
    /**
     * Read NFC tag and extract student ID
     * Expected format: {"student_id": "S12345"}
     */
    fun readNfcTag(tag: Tag): Map<String, String>? {
        return try {
            val ndef = Ndef.get(tag)
            
            if (ndef != null) {
                ndef.connect()
                
                val ndefMessage = ndef.ndefMessage
                val records = ndefMessage.records
                
                for (record in records) {
                    val payload = record.payload
                    val text = String(payload, Charsets.UTF_8).trim()
                    
                    Log.d(TAG, "NFC payload: $text")
                    
                    // Try to parse as JSON
                    if (text.startsWith("{") && text.contains("student_id")) {
                        return parseJson(text)
                    }
                    
                    // Fallback: direct student ID format
                    if (text.startsWith("S") && text.length in 5..10) {
                        return mapOf("student_id" to text)
                    }
                }
                
                ndef.close()
            }
            
            // Fallback: use tag ID
            val tagId = tag.id.joinToString("") { "%02X".format(it) }
            Log.d(TAG, "Fallback to tag ID: $tagId")
            mapOf("student_id" to "TAG_$tagId")
            
        } catch (e: Exception) {
            Log.e(TAG, "Failed to read NFC tag", e)
            null
        }
    }
    
    private fun parseJson(json: String): Map<String, String> {
        val result = mutableMapOf<String, String>()
        
        // Simple JSON parsing (avoid Gson dependency here)
        val pairs = json.trim('{', '}').split(",")
        for (pair in pairs) {
            val parts = pair.split(":").map { it.trim().trim('"') }
            if (parts.size == 2) {
                result[parts[0]] = parts[1]
            }
        }
        
        return result
    }
}
```

```python
"""
Epson ePOS Print Service
"""

import socket
import logging
from typing import Optional, Dict, Any
from dataclasses import dataclass
from enum import Enum

logger = logging.getLogger(__name__)


class PrintStatus(Enum):
    SUCCESS = "success"
    PRINTER_OFFLINE = "printer_offline"
    PAPER_OUT = "paper_out"
    ERROR = "error"


@dataclass
class PrinterConfig:
    ip: str
    port: int = 9100
    name: str = "Epson Printer"
    location: str = "Library"


class EpsonPrinter:
    """Epson network printer using ESC/POS commands"""
    
    # ESC/POS Commands
    ESC = b'\x1b'
    GS = b'\x1d'
    
    def __init__(self, config: PrinterConfig):
        self.config = config
        self.socket: Optional[socket.socket] = None
    
    def connect(self) -> bool:
        """Connect to printer"""
        try:
            self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            self.socket.settimeout(10)
            self.socket.connect((self.config.ip, self.config.port))
            return True
        except Exception as e:
            logger.error(f"Failed to connect to printer {self.config.ip}: {e}")
            return False
    
    def disconnect(self):
        """Disconnect from printer"""
        if self.socket:
            try:
                self.socket.close()
            except:
                pass
            self.socket = None
    
    def initialize(self):
        """Initialize printer"""
        self._send(b'\x1b\x40')  # ESC @
    
    def set_alignment(self, align: str = "left"):
        """Set text alignment"""
        alignments = {"left": b'\x00', "center": b'\x01', "right": b'\x02'}
        self._send(self.ESC + b'\x61' + alignments.get(align, b'\x00'))
    
    def set_bold(self, enabled: bool = True):
        """Set bold text"""
        self._send(self.ESC + b'\x45' + (b'\x01' if enabled else b'\x00'))
    
    def set_double_height(self, enabled: bool = True):
        """Set double height text"""
        self._send(self.ESC + b'\x21' + (b'\x10' if enabled else b'\x00'))
    
    def text(self, text: str, newline: bool = True):
        """Print text"""
        self._send(text.encode('utf-8'))
        if newline:
            self._send(b'\n')
    
    def line(self, char: str = "-", length: int = 32):
        """Print a line"""
        self.text(char * length)
    
    def feed(self, lines: int = 1):
        """Feed paper"""
        for _ in range(lines):
            self._send(b'\n')
    
    def cut(self):
        """Cut paper"""
        self._send(b'\n' * 3)
        self._send(self.GS + b'V\x00')  # Full cut
    
    def open_cash_drawer(self):
        """Open cash drawer (if connected)"""
        self._send(self.ESC + b'p\x00\x19\x19')
    
    def print_document(
        self,
        title: str,
        content: str,
        student_name: str,
        copies: int = 1,
        cost: int = 0
    ) -> PrintStatus:
        """Print a full document"""
        
        if not self.connect():
            return PrintStatus.PRINTER_OFFLINE
        
        try:
            for copy in range(copies):
                self.initialize()
                
                # Header
                self.set_alignment("center")
                self.set_bold(True)
                self.set_double_height(True)
                self.text("KOBEAI PRINT")
                self.set_double_height(False)
                self.set_bold(False)
                self.line()
                
                # Document info
                self.set_alignment("left")
                self.text(f"Student: {student_name}")
                self.text(f"Document: {title}")
                self.text(f"Date: {self._get_date()}")
                self.line()
                
                # Content
                self.set_alignment("left")
                for line in self._wrap_text(content, 40):
                    self.text(line)
                
                self.line()
                
                # Footer
                self.set_alignment("center")
                self.text(f"Cost: TSh {cost}")
                self.text(f"Copy {copy + 1}/{copies}")
                
                if copy < copies - 1:
                    self.feed(2)
            
            self.cut()
            return PrintStatus.SUCCESS
            
        except Exception as e:
            logger.error(f"Print failed: {e}")
            return PrintStatus.ERROR
        finally:
            self.disconnect()
    
    def print_text_file(
        self,
        file_path: str,
        student_name: str,
        cost: int = 0
    ) -> PrintStatus:
        """Print a text file"""
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()
            
            title = file_path.split('/')[-1]
            return self.print_document(title, content, student_name, cost=cost)
            
        except Exception as e:
            logger.error(f"Failed to print file {file_path}: {e}")
            return PrintStatus.ERROR
    
    def _send(self, data: bytes):
        """Send raw data to printer"""
        if self.socket:
            self.socket.send(data)
    
    def _get_date(self) -> str:
        from datetime import datetime
        return datetime.now().strftime("%Y-%m-%d %H:%M")
    
    def _wrap_text(self, text: str, width: int) -> list:
        """Wrap text to specified width"""
        words = text.split()
        lines = []
        current_line = []
        current_length = 0
        
        for word in words:
            if current_length + len(word) + 1 <= width:
                current_line.append(word)
                current_length += len(word) + 1
            else:
                if current_line:
                    lines.append(' '.join(current_line))
                current_line = [word]
                current_length = len(word)
        
        if current_line:
            lines.append(' '.join(current_line))
        
        return lines if lines else [text]


# Printer registry
PRINTERS = {
    "library": PrinterConfig(
        ip="192.168.1.50",
        name="Library Printer",
        location="School Library"
    ),
    "staff_room": PrinterConfig(
        ip="192.168.1.51",
        name="Staff Room Printer",
        location="Staff Room"
    ),
}


def get_printer(printer_id: str = "library") -> Optional[EpsonPrinter]:
    """Get printer by ID"""
    config = PRINTERS.get(printer_id)
    if config:
        return EpsonPrinter(config)
    return None
```

```python
    request: NfcAuthRequest,
    db: AsyncSession = Depends(get_db)
):
    """Authenticate student via NFC"""
    
    # Find student
    from app.models.user import User
    from sqlalchemy import select
    
    result = await db.execute(
        select(User).where(User.student_id_number == request.student_id)
    )
    student = result.scalar_one_or_none()

        "student_id": student.student_id_number,
        "name": student.full_name
    })

@router.post("", response_model=PrintResponse)
async def print_document(
    request: PrintRequest,
    token: str = Depends(lambda t: t),
    db: AsyncSession = Depends(get_db)
):
    """Print a document"""
    
    payload = security_service.verify_token(token)
    if not payload:
        raise HTTPException(401, "Invalid token")
    
    service = PrintService(db)
    
    # Get document
    document = await service.get_document(UUID(request.document_id))
    if not document:
        raise HTTPException(404, "Document not found")
    
    # Calculate cost
    total_cost = document.print_cost * request.copies
    
    # Check balance
    balance_ok = await service.check_balance(request.student_id, total_cost)
    if not balance_ok:
        return PrintResponse(
            success=False,
            message=f"Insufficient balance. Need TSh {total_cost}"
        )
    
    # Create print job
    job = await service.create_print_job(
        student_id=request.student_id,
        document_id=request.document_id,
        copies=request.copies,
        cost=total_cost
    )
    
    # Deduct from balance
    new_balance = await service.deduct_cost(request.student_id, total_cost, job.id)
    
    # Send to printer
    printer = get_printer("library")
    if printer:
        content = await service.get_document_content(document.id)
        status = printer.print_document(
            title=document.title,
            content=content,
            student_name=payload.get("name", "Student"),
            copies=request.copies,
            cost=total_cost
        )
        
        if status == PrintStatus.SUCCESS:
            await service.mark_job_completed(job.id)
            return PrintResponse(
                success=True,
                message="Document printed successfully!",
                job_id=str(job.id),
                new_balance=new_balance
            )
        else:
            # Refund
            await service.refund_cost(request.student_id, total_cost, job.id)
            return PrintResponse(
                success=False,
                message=f"Printer error: {status.value}"
            )
    else:
        return PrintResponse(
            success=False,
            message="Printer not available"
        )
```
